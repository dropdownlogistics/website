RAG Bridge dex-bridge.py
# Standard query
python dex-bridge.py "your query here"
# More chunks
python dex-bridge.py "your query here" --top 10
# Search RAW archive instead of canon
python dex-bridge.py "your query here" --raw
# Include external collections
python dex-bridge.py "your query here" --external
# Specific model
python dex-bridge.py "your query here" --model llama3.1:8b
# Verbose — show retrieved chunks with scores
python dex-bridge.py "your query here" --verbose
# Route to laptop node
python dex-bridge.py "your query here" --node laptop
# Skip auto-ingest (calibration runs)
python dex-bridge.py "your query here" --no-ingest
# Interactive mode
python dex-bridge.py --interactive
Weighted Query dex_weights.py
# Query with weights
python dex_weights.py "your query here"
# More results
python dex_weights.py "your query here" --n 10
# Include external collections
python dex_weights.py "your query here" --external
# Show weight table
python dex_weights.py --stats
Ingestion dex-ingest.py
# Standard ingest — RAW + CANON routing
python dex-ingest.py --path "C:\path\to\folder"
# Canon only (does not expand RAW)
python dex-ingest.py --path "C:\path\to\folder" --build-canon
# Scoped collection
python dex-ingest.py --path "C:\path\to\folder" --collection dex_code
# Extension filter
python dex-ingest.py --path "C:\path\to\folder" --ext-filter .py .cs .ts
# Tag nominating seat
python dex-ingest.py --path "C:\path\to\folder" --collection ext_creator --nominated-by "1006"
# Fast mode (skip ID load — for auto-ingest paths)
python dex-ingest.py --path "C:\path\to\folder" --build-canon --fast
# Reset collections (destructive)
python dex-ingest.py --path "C:\path\to\folder" --reset
# Single file ingest
Copy-Item "C:\path\to\file.txt" "C:\Users\dexjr\dex-rag\ingest-single\" python dex-ingest.py --path "C:\Users\dexjr\dex-rag\ingest-single" --build-canon
Council dex-council.py
# All models + RAG + save + ingest
python dex-council.py --from-file "C:\Users\dexjr\dex-rag\prompts\PROMPT.txt" --all --rag --save council-runs/NAME --ingest
# Local models only + RAG
python dex-council.py "prompt text here" --rag
# Cloud models only
python dex-council.py --from-file "prompt.txt" --cloud-only --rag
# Custom timeout
python dex-council.py --from-file "prompt.txt" --all --rag --timeout 240
# Re-synthesize from saved folder
python dex-council.py --synthesize council-runs/NAME
# Retry failed models + verbose
python dex-council.py --from-file "prompt.txt" --all --rag --retry --verbose
